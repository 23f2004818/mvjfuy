import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import mpld3
import inspect

# ----- Step 1. Generate synthetic employee dataset -----
data = {
    "EmployeeID": range(1, 101),
    "Department": [
        "Operations", "HR", "Finance", "Sales", "IT"
    ] * 20,  # repeating pattern for 100 employees
    "Region": ["North", "South", "East", "West", "Central"] * 20
}

df = pd.DataFrame(data)

# ----- Step 2. Frequency count for Operations -----
operations_count = (df["Department"] == "Operations").sum()
print("Number of employees in Operations:", operations_count)

# ----- Step 3. Create histogram of departments -----
plt.figure(figsize=(8, 6))
sns.countplot(data=df, x="Department", palette="viridis")
plt.title("Department Distribution of Employees")
plt.xlabel("Department")
plt.ylabel("Frequency")

# Convert chart to HTML
chart_html = mpld3.fig_to_html(plt.gcf())

# ----- Step 4. Embed Python source code -----

    source_code = f.read()

"""
